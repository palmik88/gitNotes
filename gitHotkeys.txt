git init - запустить гит в текущей директории

git add file.txt - добавить файл
git add * - добавить все файлы из директории

git checjout --file.txt - отменить все изменения в файле

git status - проверить статус

git config -l - посмотреть настройки
git config --global user.name "Mikhail Palgunov" - задать имя
git config --global user.email "qqq@www.rrr" - задать email

git log - посмотреть все коммиты
git log -1 - посмотреть последний коммит
git log -1 -p - посмотреть последний коммит подробнее

git diff --staged -Показывает что будет записано в git.log, когда файл в статусе Ready for commit (зеленый)

nano .gitignor - создать файл, в котором бдует прописано то, что будет игнорироваться
Пример: *.log - все файлы , заканчивающиеся на .log
	logs/ - будет игнорироваться директория
	file.txt - будет игнорироваться конкретный файл

git clone https://github.com/palmik88/gitNotes.git - скопировать из репозитория


git remote -v: показывает ссылку на репозиторий.
Если ничего не выдаёт, то:
git remote add origin https://github.com/palmik88/gitNotes.git -добавляем ссылку
git remote set-url origin git@github.com:palmik88/gitNotes.git - меняет ссылку


Запушить на гит: -f - запушить принудительно
git push -f https://github.com/palmik88/gitNotes.git master
